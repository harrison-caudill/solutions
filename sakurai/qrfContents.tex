\section{General Purpose}

\subsection{Core Concepts}

\begin{equation}
  \tag{1.210}
  \vectorbold{x}_{new}
  \equiv \vectorbold{X}
  = \vectorbold{x} + d\vectorbold{x},
  \vectorbold{p}_{new}
  \equiv \vectorbold{P}
  = \vectorbold{p}
\end{equation}

\begin{equation}
  \tag{1.211}
  F(\vectorbold{x}, \vectorbold{P})
  = \vectorbold{x} \cdot \vectorbold{P}
  + \vectorbold{p} \cdot d\vectorbold{x}
\end{equation}


\subsection{Trig Identities}

\begin{equation}
  \sin{2\alpha} = 2\sin{\alpha}\cos{\alpha}
\end{equation}

\begin{equation}
  \begin{split}
    \qlabel{trigCos2A}
    \cos{2\alpha}
    &= \cos^2{\alpha} - \sin^2{\alpha}\\
    &= 2\cos^2{\alpha} - 1\\
    &= 1 - 2\sin^2{\alpha}\\
  \end{split}
\end{equation}

\begin{equation}
  \cos{\frac \alpha 2} = \pm \sqrt{\frac{1 + \cos{\alpha}}{2}}
\end{equation}

\begin{equation}
  \sin{\frac \alpha 2} = \pm \sqrt{\frac{1 - \cos{\alpha}}{2}}
\end{equation}

\section{Pauli Spin}

\begin{equation*}
  \tag{1.110a}
  \ket{x\pm} = \frac{1}{\sqrt{2}} [\ket + \pm \ket -]
\end{equation*}

\begin{equation*}
  \tag{1.110b}
  \ket{y\pm} = \frac{1}{\sqrt{2}} [\ket + \pm i\ket -]
\end{equation*}

\begin{equation*}
  \tag{1.111a}
  S_x = \frac{\hbar}{2} [\ket - \bra + + \ket + \bra - ]
\end{equation*}

\begin{equation*}
  \tag{1.111b}
  S_y = i \frac{\hbar}{2} [\ket - \bra + - \ket + \bra - ]
\end{equation*}

\begin{equation*}
  \tag{3.50}
  \sigma_x =
  \begin{pmatrix}
    0 & 1 \\
    1 & 0 \\
  \end{pmatrix},
  \sigma_y =
  \begin{pmatrix}
    0 & -i \\
    i &  0 \\
  \end{pmatrix},
  \sigma_z =
  \begin{pmatrix}
    1 &  0 \\
    0 & -1 \\
  \end{pmatrix}
\end{equation*}

\subsection{Integral Table}

\begin{equation}
  \qlabel{integral:gaussian}
  \int_{-\infty}^{\infty} dx\  e^{-\alpha x^2} = \sqrt{\frac{\pi}{\alpha}}
\end{equation}

\begin{equation}
  \qlabel{integral:xgaussian}
  \int_{-\infty}^{\infty} dx\  xe^{-\alpha x^2} = 0
\end{equation}\cite[eq 3.321.3]{gr}

\begin{equation}
  \qlabel{integral:xxgaussian}
  \int_{-\infty}^{\infty} dx\  x^2e^{-\alpha x^2} = \frac{1}{2\alpha}\sqrt{\frac{\pi}{\alpha}}
\end{equation}

\section{Wave Functions}

\subsection{\acf{gwp}}
\qlabel{wavey:gwp}
    \begin{equation}
      \inlabel{gwp:exp}
      \begin{alignedat}{2}
        \expval{X}^2 &= 0 & \quad{} &\text{(eq \ref{1.24:exp:x1})}\\
        \expval{X^2} &= \frac{d^2}{2} & &\text{(eq \ref{1.24:exp:x2})}\\
        \expval{P}^2 &= \hbar^2k^2 & &\text{(eq \ref{1.24:exp:p1})}\\
        \expval{P^2} &= \hbar^2\left(k^2+\frac{1}{2d^2}\right)
        & &\text{(eq \ref{1.24:exp:p2})}\\
      \end{alignedat}
    \end{equation}

    \begin{equation}
      \dispersion{X}\dispersion{P}
      =
      \left(\expval{\hat{X}^2} - \expval{\hat{X}}^2\right)
      \left(\expval{\hat{P}^2} - \expval{\hat{P}}^2\right)
      = \frac{\hbar^2}{4}
    \end{equation}

    \begin{equation}
      \qlabel{wavey:gwp:psi}
      \psi(x')
      = \braket{x'}{\alpha}
      = \frac{1}{\pi^{1/4}}\sqrt{\frac{1}{d}}
      \exp[-\frac{1}{2d^2}x'^{2}]
    \end{equation}

    \begin{equation}
      \qlabel{wavey:gwp:phi}
      \phi(p')
      =
      \left[ \frac{1}{\sqrt{2\pi\hbar}} \right]
      \int dx' \exp(\frac{-ip'x'}{\hbar})\psi_{\alpha}(x')
      = 
      \frac{1}{\pi^{1/4}}
      \sqrt{\frac{d}{\hbar}}
      \exp[-\frac{d^2}{2\hbar^2}p^2]
    \end{equation}
