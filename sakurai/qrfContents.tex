  \qrfheader{Eigen Decomposition}

  \qrfeq[decomp:ident]{
    I = \sum_i \Lambda_i \implies I = \int \dd{a'} \Lambda_{a'}
  }{\cite[1.184]{sakurai}}

  \qrfeq[decomp:func]{
    f(A) = \int \dd{a'} f(a') \Lambda_{a'}
  }{\outref{1}{29}{mainPoint}}


  
  \qrfheader{General}

  \qrfeq[taylor:exponential]{
    e^{A} \equiv \sum_{k=0}^{\infty} \frac{1}{k!} A^k
  }{\cite[1.219]{sakurai}}

  \qrfeq[stats:var:minus]{
    \dispersion{A} = \expval{A^2} - \expval{A}^2
  }{\cite[1.144]{sakurai}}

  \qrfeq[stats:var:func]{
    \dispersion{f(X)} \approx \left(f'(\expval{X})\right)^2 \dispersion{X}
  }{\cite{1.24:wikipedia:Variance}}


  \qrfheader{Uncertainty Relations}

  \qrfeq[uncertain:general]{
    \dispersion{A}\dispersion{B} \ge \frac{1}{4} \abs{\expval{[A,B]}}^2
  }{\cite[1.146]{sakurai}}

  \qrfeq[uncertain:xp]{
    [x,p] = i\hbar
  }{\cite[1.215]{sakurai}}

  \qrfeq[uncertain:et]{
    \Delta E \Delta t \gtrsim \hbar/2
  }{\cite{1.24:wikipedia:Uncertainty}}

  \qrfheader{Pauli Spin}

  \qrfeq[spin:kets]{
    \begin{split}
      \ket{x\pm} &= \frac{1}{\sqrt{2}} [\ket + \pm \ket -]\\
      \ket{y\pm} &= \frac{1}{\sqrt{2}} [\ket + \pm i\ket -]\\
    \end{split}
  }{\cite[1.110]{sakurai}}

  \qrfeq[spin:sxy]{
    \begin{split}
      S_x &= \frac{\hbar}{2} [\ket - \bra + + \ket + \bra - ]\\
      S_y &= i \frac{\hbar}{2} [\ket - \bra + - \ket + \bra - ]\\
    \end{split}
  }{\cite[1.111]{sakurai}}

  \qrfeq[spin:matrices]{
    \begin{split}
      \sigma_x &=
      \begin{pmatrix}
        0 & 1 \\
        1 & 0 \\
      \end{pmatrix},
      \\
      \sigma_y &=
      \begin{pmatrix}
        0 & -i \\
        i &  0 \\
      \end{pmatrix},
      \\
      \sigma_z &=
      \begin{pmatrix}
        1 &  0 \\
        0 & -1 \\
      \end{pmatrix}
      \\
    \end{split}
  }{\cite[3.50]{sakurai}}

  \qrfeq[spin:angle]{
    \ket{\alpha} =
    \cos(\frac{\beta}{2})\ket{+} + \sin(\frac{\beta}{2}) e^{i\alpha}\ket{-}
  }{\outref{1}{11}{spinAngle}}

  \qrfheader{Table of Integrals}

  \qrfeq[integral:gaussian]{
    \int_{-\infty}^{\infty} \dd{x} e^{-\alpha x^2} = \sqrt{\frac{\pi}{\alpha}}
  }{\cite[eq 3.321.3]{gr}}

  \qrfeq[integral:xgaussian]{
    \int_{-\infty}^{\infty} \dd{x} xe^{-\alpha x^2} = 0
  }{\cite[eq 3.321.4]{gr}}

  \qrfeq[integral:xxgaussian]{
    \int_{-\infty}^{\infty} \dd{x} x^2e^{-\alpha x^2}
    = \frac{1}{2\alpha}\sqrt{\frac{\pi}{\alpha}}
  }{\cite[eq FIXME]{gr}}


  \qrfheader{Commutations}

  \qrfeq[comms:basic]{
    \begin{split}
      [A,A] &= 0\\
      [A,B] &= -[B,A]\\
      [A,c] &= 0 \text{ (c is just a number)}\\
      [A+B,C] &= [A,C] + [B,C]\\
      [A,BC] &= [A,B]C + B[A,C]\\
    \end{split}
  }{\cite[1.232]{sakurai}}

  \qrfeq[comms:power]{
    [A,B^n] = n B^{n-1}\underbrace{[A,B]}_{[A,B] \in \mathbb{C}}
  }{\outref{1}{30}{ConstantPower}}

  \qrfeq[comms:func]{
    [x, F(p)] = i\hbar \pdv{F}{p}, \quad [p, F(x)] = -i\hbar \pdv{F}{x}
  }{\outref{1}{30}{xfp}}


  
  \qrfheader{Gaussian Wave Packets}
  \qrfeq[wavey:gwp:exp]{
    \begin{alignedat}{2}
      \expval{X}^2 &= 0\\
      \expval{X^2} &= \frac{d^2}{2}\\
      \expval{P}^2 &= \hbar^2k^2\\
      \expval{P^2} &= \hbar^2\left(k^2+\frac{1}{2d^2}\right)\\
    \end{alignedat}
  }{\outref{1}{24}{exp:x1},
    \outref{1}{24}{exp:x2},
    \outref{1}{24}{exp:p1},
    \outref{1}{24}{exp:p2}}


  \qrfeq[wavey:gwp:psi]{
    \begin{split}
      \psi(x')
      &= \braket{x'}{\alpha}\\
      &= \frac{1}{\pi^{1/4}}\sqrt{\frac{1}{d}}
      \exp[ikx'-\frac{x'^2}{2d^2}]\\
    \end{split}
  }{\cite[eq 1.267]{sakurai}}

  \qrfeq[wavey:gwp:phi]{
    \begin{split}
      \phi(p')
      &=
      \left[ \frac{1}{\sqrt{2\pi\hbar}} \right]
      \int \dd{x'} \exp(\frac{-ip'x'}{\hbar})\psi_{\alpha}(x')
      \\
      &= 
      \frac{1}{\pi^{1/4}}
      \sqrt{\frac{d}{\hbar}}
      \exp[-\frac{d^2}{2\hbar^2}(p' - \hbar k)^2]\\
    \end{split}
  }{\cite[eq 1.274]{sakurai}}



  \qrfheader{Trig Identities}
  
  \qrfeq[trig:sin2a]{
    \sin{2\alpha} = 2\sin{\alpha}\cos{\alpha}
  }{}

  \qrfeq[trig:cos2a]{
    \begin{split}
      \cos{2\alpha}
      &= \cos^2{\alpha} - \sin^2{\alpha}\\
      &= 2\cos^2{\alpha} - 1\\
      &= 1 - 2\sin^2{\alpha}\\
    \end{split}
  }{}

  \qrfeq[trig:coshalf]{
    \cos{\frac \alpha 2} = \pm \sqrt{\frac{1 + \cos{\alpha}}{2}}
  }{}

  \qrfeq[trig:sinhalf]{
    \sin{\frac \alpha 2} = \pm \sqrt{\frac{1 - \cos{\alpha}}{2}}
  }{}

  \qrfeq[trig:harmonic]{
    \begin{alignedat}{2}
      && f(\theta) &= a\cos(\theta) + b\sin(\theta)\\
      &&&= c\cos(\theta + \delta)\\
      \text{where } && c &= \sqrt{a^2 + b^2}\\
      \text{and } && \delta &= \tan^{-1}(-\frac{a}{b})
    \end{alignedat}
  }{\cite{1.24:msu:harmonicAddition}}



  \qrfheader{Operators}
    \qrfeq[misc:realeig]{

    \qrfeq[misc:unitary]{
      U^{\dagger}U = I \implies \text{U is Unitary}
    }{\cite[eq 1.158]{sakurai}}

  \qrfeq[op:p]{
    \mel{x'}{\hat{p}}{\alpha} = -i\hbar\pdv{}{x'}\braket{x'}{\alpha}
  }{\cite[eq 1.249]{sakurai}}

  \qrfeq[op:x]{
    \mel{p'}{\hat{x}}{\alpha} = i\hbar\pdv{}{p'}\braket{p'}{\alpha}
  }{\outref{1}{36}{theQuestion}}
