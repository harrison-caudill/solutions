\chapter{Quick Reference}

Because the $'$ and $''$ and $'''$ and so on gets really annoying, I
switch from primed Eigenvectors/Eigenvalues to subscripts.  So you'll
see $\ket{a'} \Rightarrow \ket{a_i}$ and $\ket{a''} \Rightarrow
\ket{a_j}$ and so on.


\section{General Purpose}

\subsection{Core Concepts}

\begin{equation}
  \tag{1.210}
  \vectorbold{x}_{new}
  \equiv \vectorbold{X}
  = \vectorbold{x} + d\vectorbold{x},
  \vectorbold{p}_{new}
  \equiv \vectorbold{P}
  = \vectorbold{p}
\end{equation}

\begin{equation}
  \tag{1.211}
  F(\vectorbold{x}, \vectorbold{P})
  = \vectorbold{x} \cdot \vectorbold{P}
  + \vectorbold{p} \cdot d\vectorbold{x}
\end{equation}


\subsection{Trig Identities}

\begin{equation}
  \sin{2\alpha} = 2\sin{\alpha}\cos{\alpha}
\end{equation}

\begin{equation}
  \begin{split}
  \cos{2\alpha}
  &= \cos^2{\alpha} - \sin^2{\alpha}
  &= 2\cos^2{\alpha} - 1
  &= 1 - 2\sin^2{\alpha}
  \end{split}
\end{equation}

\begin{equation}
  \cos{\frac \alpha 2} = \pm \sqrt{\frac{1 + \cos{\alpha}}{2}}
\end{equation}

\begin{equation}
  \sin{\frac \alpha 2} = \pm \sqrt{\frac{1 - \cos{\alpha}}{2}}
\end{equation}

\section{Pauli Spin}

\begin{equation*}
  \tag{1.110a}
  \ket{x\pm} = \frac{1}{\sqrt{2}} [\ket + \pm \ket -]
\end{equation*}

\begin{equation*}
  \tag{1.110b}
  \ket{y\pm} = \frac{1}{\sqrt{2}} [\ket + \pm i\ket -]
\end{equation*}

\begin{equation*}
  \tag{1.111a}
  S_x = \frac{\hbar}{2} [\ket - \bra + + \ket + \bra - ]
\end{equation*}

\begin{equation*}
  \tag{1.111b}
  S_y = i \frac{\hbar}{2} [\ket - \bra + - \ket + \bra - ]
\end{equation*}

\begin{equation*}
  \tag{3.50}
  \sigma_x =
  \begin{pmatrix}
    0 & 1 \\
    1 & 0 \\
  \end{pmatrix},
  \sigma_y =
  \begin{pmatrix}
    0 & -i \\
    i &  0 \\
  \end{pmatrix},
  \sigma_z =
  \begin{pmatrix}
    1 &  0 \\
    0 & -1 \\
  \end{pmatrix}
\end{equation*}
