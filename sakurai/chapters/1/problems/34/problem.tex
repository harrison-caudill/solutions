\begin{question}

  \begin{questionpart}
    Starting with a momentum operator $\vb{p}$ having eigenstates
    $\ket{\vb{p}'}$, define an infinitesimal boost operator
    $\mathscr{B}(d\vb{p}')$ that changes one momentum eigenstate into
    another, that is

    \begin{equation*}
      \mathscr{B}(d\vb{p}')\ket{\vb{p}'} = \ket{\vb{p}' + d\vb{p}'}
    \end{equation*}
  \end{questionpart}

  \begin{questionpart}
    Show that the form $\mathscr{B}(d\vb{p}') = 1 + i\vb{W}\cdot
    d\vb{p}'$, where W is Hermitian, satisfies the unitary,
    associative, and inverse properties that are appropriate for
    $\mathscr{B}(d\vb{p}')$.
  \end{questionpart}

  \begin{questionpart}
    Use dimensional analysis to express $\vb{W}$ in terms of the
    position operator $\vb{x}$, and show that the result satisfies the
    canonical commutation relations $[x_i,p_j] = -i\hbar\delta_{ij}$.
  \end{questionpart}

  \begin{questionpart}
    Derive an expression for the matrix element
    $\mel{\vb{p}'}{\vb{x}}{\alpha}$ in terms of a derivative with
    respect to $\vb{p}'$ of $\braket{\vb{p}'}{\alpha}$.
  \end{questionpart}
\end{question}
