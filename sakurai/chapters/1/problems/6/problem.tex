\begin{question}

  \begin{questionpart}
    $tr(XY) = tr(YX)$, where $X$ and $Y$ are operators
  \end{questionpart}

  
  \begin{questionpart}
    $(XY)^\dagger =Y^\dagger X^\dagger$, where $X$ and $Y$ are operators
  \end{questionpart}

  \begin{questionpart}
    $exp[i\ f(A)] = ?$ in ket-bra form, where $A$ is a Hermitian
    operator whose eigenvalues are known;
  \end{questionpart}

  \begin{questionpart}
    $\sum_{a'} \psi_{a'}^*(\vectorbold{x'}) \psi_{a'}(\vectorbold{x''})$,
    where $\psi_{a'}(\vectorbold{x'}) = \bra{x'}\ket{a'}$
  \end{questionpart}

\end{question}
