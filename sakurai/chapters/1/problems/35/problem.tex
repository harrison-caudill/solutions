\begin{question}
  \begin{questionpart}
    Verify (1.271a) and (1.271b) for the expectation value of $p$ and
    $p^2$ from the Gaussian wave packet (1.267).
  \end{questionpart}

  \begin{questionpart}
    Evaluate the expectation value of $p$, and $p^2$ using the
    momentum-space wave function (1.274).
  \end{questionpart}
\end{question}

\end{purpose}

\begin{answer}
  \begin{answerpart}{Proving P}
    \begin{equation}
      \tag{1.271a}
      \expval{p} = \hbar k
    \end{equation}

    Reference problem \ref{1.24:theQuestion} equation
    \ref{1.24:exp:p1}.
    
    \begin{equation}
      \tag{1.271b}
      \expval{p^2} = \frac{\hbar^2}{2d^2} + \hbar^2k^2
    \end{equation}

    Reference problem \ref{1.24:theQuestion} equation
    \ref{1.24:exp:p2}.
  \end{answerpart}

  \begin{answerpart}{Fourier Was Right}
    \begin{equation}
      \tag{1.274}
      \begin{split}
        \braket{p'}{\alpha}
        &=
        \left(\frac{1}{\sqrt{2\pi\hbar}}\right)
        \left(\frac{1}{\pi^{1/4}\sqrt{d}}\right)
        \int_{-\infty}^{\infty} dx'\
        \exp(\frac{-ip'x'}{\hbar} + ikx' - \frac{x^2}{2d^2})
        \\
        &= \sqrt{\frac{d}{\hbar\sqrt{\pi}}}
        \exp[\frac{-(p' - \hbar k)^2d^2}{2\hbar^2}]
        \\
      \end{split}
    \end{equation}

    \begin{equation}
      \begin{split}
        \expval{p}
        &= \mel{\alpha}{p}{\alpha}\\
        &= \int_{-\infty}^{\infty} dp' p' \braket{\alpha}{p'}\braket{p'}{\alpha}\\
        &= \int_{-\infty}^{\infty} dp' p' \braket{p'}{\alpha}^{*}\braket{p'}{\alpha}
        \\
        &=
        \frac{d}{\hbar\sqrt{\pi}}
        \int_{-\infty}^{\infty} dp' p'
        \exp[\frac{-(p' - \hbar k)^2d^2}{\hbar^2}]
        \\
        &=
        \frac{d}{\hbar\sqrt{\pi}}
        \int_{-\infty}^{\infty} dp' p'
        \exp[\frac{-d^2}{\hbar^2}\left(
          p'^2 - 2\hbar k p' + \hbar^2k^2
          \right)]
        \\
        &=
        \frac{d}{\hbar\sqrt{\pi}}
        \int_{-\infty}^{\infty} dp' p'
        \exp[AB^2]
        \\
        &= 0\\
      \end{split}
    \end{equation}
    
  \end{answerpart}

\end{answer}
