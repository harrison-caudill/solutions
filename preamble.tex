%% Because we have a bunch of tagged equations when we reference
%% equations from the book, you can sometimes get a duplicate warning
%% for an internal label that hyperref uses.  While it works itself
%% out, it also spams the log with warning lines.  Here, we explicitly
%% take control of the hyperref counter
\newcounter{HAnchorEq}
\AddToHook{env/equation/begin}{\stepcounter{HAnchorEq}}
