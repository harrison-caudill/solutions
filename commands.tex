%% NOTE: The places where I have \theHequation is so that the internal
%% reference infrastructure employed by the hyperref package doesn't
%% experience any name collisions.  You don't actually interact with
%% it, so it just needs to be unique.


\NewDocumentCommand{\appendixCounters}{}{%
    % In appendices, we want our equations/figures to increment within
    % the chapters, not any other level.
    \counterwithin{equation}{chapter}
    \counterwithout{equation}{section}
    \counterwithout{equation}{subsection}
    \renewcommand{\theequation}{\thechapter.\arabic{equation}}
    \renewcommand{\theHequation}{\thechapter.\arabic{equation}.\arabic{HAnchorEq}}
    %
    \counterwithin{figure}{chapter}
    \counterwithout{figure}{section}
    \counterwithout{figure}{subsection}
    \renewcommand{\thefigure}{\thechapter.\arabic{figure}}
    \setcounter{chapter}{0}
}%

\NewDocumentCommand{\problemCounters}{}{%
    % In problems, we want our equations/figures to increment within
    % the chapters, not any other level.
    \counterwithin{equation}{chapter}
    \counterwithin{equation}{section}
    \counterwithout{equation}{subsection}
    \counterwithout{equation}{subsubsection}
    \renewcommand{\theequation}{\thesection.\arabic{equation}}
    \renewcommand{\theHequation}{\thechapter.\arabic{equation}.\arabic{HAnchorEq}}
    %
    \counterwithin{figure}{chapter}
    \counterwithin{figure}{section}
    \counterwithout{figure}{subsection}
    \counterwithout{figure}{subsubsection}
    \renewcommand{\thefigure}{\thesection.\arabic{figure}}
    %
    \renewcommand{\thechapter}{\arabic{chapter}}
    \setcounter{chapter}{1}
}%
%% In the main document, import a chapter of problem solutions.
\NewDocumentCommand{\probChapter}{mm}{%

    % We may or may not work all chapters in sequence, so we
    % explicitly assign the 
    \setcounter{chapter}{\numexpr #1 - 1 \relax}%

    \chapter{#2}%
    \input{\bookName/chapters/#1/chapter}%
}%

\NewDocumentCommand{\problem}{mm}{%
    \setcounter{section}{\numexpr #1-1 \relax}%
    \refstepcounter{section}%
    \needspace{5\baselineskip}
    \phantomsection
    \section*{\thesection: #2}
    \inlabel{theQuestion}
    \addcontentsline{toc}{section}{%
    \texorpdfstring{\thesection:\ #2}{\thesection:\ #2}%
    }%
    \graphicspath{%
        {\buildPath/\bookName}%
        {\buildPath/\bookName/chapters/\thechapter/problems/#1}%
    }%
    \acresetall
    \input{\bookName/chapters/\thechapter/problems/#1/problem}
}

\newenvironment{question}{%
    \subsection*{Recitation of the Problem}
    \begin{quote}
    \raggedright % when the text is terse, you don't want underfull hbox's
    \noindent
}{%
    \end{quote}
}

\newenvironment{answer}{%
\setcounter{qpart}{0}
}{%
}%

\newenvironment{answerpart}[1]{%
    \renewcommand{\theHqpart}{A.\thesection.\arabic{qpart}}
    \setcounter{qspart}{0}
    \refstepcounter{qpart}
    \renewcommand{\thesubsection}{\thesection.\alph{qpart}}
    \phantomsection
    \subsection*{\thesubsection: #1}
    \addcontentsline{toc}{subsection}{%
        \texorpdfstring{\thesubsection:\ #1}{\thesubsection:\ #1}%
    }%
}{%
}

\newenvironment{answerspart}{%
    \renewcommand{\theHqspart}{A.\thesection.\arabic{qpart}.\arabic{qspart}}
    \refstepcounter{qspart}
    \renewcommand{\thesubsubsection}{\thesection.\alph{qpart}.\roman{qspart}}
    \phantomsection
    \subsubsection*{\thesubsubsection}
    \addcontentsline{toc}{subsubsection}{%
        \texorpdfstring{\thesubsubsection}{\thesubsubsection}%
    }%
}{%
}

\newenvironment{questionpart}{%
    \renewcommand{\theHqpart}{Q.\thesection.\arabic{qpart}}
    \setcounter{qspart}{0}
    \refstepcounter{qpart}
    \list{\textbf{\alph{qpart})}}{%
        \setlength{\leftmargin}{2em}   % Indentation of the text
        \setlength{\labelwidth}{1.5em} % Width of the "a)" part
        \setlength{\labelsep}{0.5em}   % Space between "a)" and text
        \setlength{\topsep}{0.5em}     % Vertical space above this part
        \setlength{\parsep}{0pt}
    }%
    \item \relax % Create the item
    %% \textbf{\alph{qpart})}\
}{%
    \endlist%
}

\newenvironment{questionsubpart}{%
    \renewcommand{\theHqspart}{Q.\thesection.\arabic{qpart}.\arabic{qspart}}
    \refstepcounter{qspart}
    \list{\textbf{\roman{qspart})}}{%
        \setlength{\leftmargin}{2em}   % Indentation of the text
        \setlength{\labelwidth}{1.5em} % Width of the "a)" part
        \setlength{\labelsep}{0.5em}   % Space between "a)" and text
        \setlength{\topsep}{0.5em}     % Vertical space above this part
        \setlength{\parsep}{0pt}
    }%
    \item \relax % Create the item
    %% \textbf{\alph{qpart})}\
}{%
    \endlist%
}

% Generates a label that is prefixed with the subsection, that way we
% can reference equations across problems if we want, but also can
% safely have namespace collisions between sections.
\newcommand{\inlabel}[1]{%
  \label{\thesection:#1}%
}%

% The corresponding ref command for the \inlabel command
\newcommand{\inref}[1]{%
  \ref{\thesection:#1}%
}%

% The corresponding ref command for the \inlabel command
\NewDocumentCommand{\incite}{o m}{%
  \IfValueTF{#1}{%
    \cite[#1]{\thesection:#2}%
  }{%
    \cite{\thesection:#2}%
  }%
}%

\newcommand{\outref}[3]{%
  \ref{#1.#2:#3}%
}%

% Similar to inlabel but meant for use with the QRF
\newcommand{\qlabel}[1]{%
  \label{q:#1}%
}%

% The corresponding ref command for the \inlabel command
\newcommand{\qref}[1]{%
  \ref{q:#1}%
}%

\NewDocumentCommand{\disputed}{m}{%
\begin{center}%
\Huge Disputed%
\end{center}%
%
#1%
}%
