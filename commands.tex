\NewDocumentCommand{\probChapter}{m}{
\setcounter{section}{#1-1}
\section{Chapter #1}
\input{\bookName/chapters/#1/chapter}
}

\NewDocumentCommand{\problem}{mm}{%
\setcounter{subsection}{#2-1}
\subsection{Problem #2}
\graphicspath{{\buildPath/\bookName/chapters/#1/problems/#2}}
\input{\bookName/chapters/#1/problems/#2/problem}
}

\newenvironment{question}{%
\begin{quote}
\noindent
}{%
\end{quote}
}

\newenvironment{answer}{%
}{%
}%

\newenvironment{answerpart}[1]{%
\subsubsection{#1}
}{%
}

\newenvironment{questionpart}{%
    \refstepcounter{qpart}
    \list{\textbf{\alph{qpart})}}{%
        \setlength{\leftmargin}{2em}   % Indentation of the text
        \setlength{\labelwidth}{1.5em} % Width of the "a)" part
        \setlength{\labelsep}{0.5em}   % Space between "a)" and text
        \setlength{\topsep}{0.5em}     % Vertical space above this part
        \setlength{\parsep}{0pt}
    }%
    \item \relax % Create the item
    %% \textbf{\alph{qpart})}\ 
}{%
    \endlist%
}

}{%
}

% Set up section/subsection/subsubsection numbering to account for
% problem naming
\renewcommand{\thesubsubsection}{\thesubsection.\alph{subsubsection}}

% Generates a label that is prefixed with the subsection, that way we
% can reference equations across problems if we want, but also can
% safely have namespace collisions between sections.
\newcommand{\inlabel}[1]{%
  \label{\thesubsection:#1}%
}%

% The corresponding ref command for the \inlabel command
\newcommand{\inref}[1]{%
  \ref{\thesubsection:#1}%
}%

\NewDocumentCommand{\disputed}{m}{%

\end{center}%

}%
